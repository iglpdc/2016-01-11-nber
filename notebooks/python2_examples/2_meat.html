What if we want to examine the series by decade?

Timestamps do not have a .decade attribute.

Let's define a function to group by decade

In [31]:
def floor_decade(date_value):
    "Takes a date. Returns the decade."
    return (date_value.year // 10) * 10
In [32]:
print floor_decade(pd.to_datetime('2015-01-08'))
print floor_decade(pd.to_datetime('1974-01-08'))
print floor_decade(pd.to_datetime('1944-01-08'))

def hist_and_kde(s2p):
    """Plot histogram and kde overlay"""
    import matplotlib.pyplot as plt
    plt.figure()
    s2p.hist(bins=31, normed=True, color="Blue")
    s2p.plot(kind='kde', linewidth=3, title=s2p.name, alpha=0.6)
    plt.show()
